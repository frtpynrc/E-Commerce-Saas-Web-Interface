var app = new function() {
  // Reference to the HTML element displaying tasks
  this.el = document.getElementById('tasks');
  // Array to store tasks
  this.tasks = [];

  // Function to fetch and display all tasks
  this.FetchAll = function() {
    var data = '';
    if (this.tasks.length > 0) {
      for (i = 0; i < this.tasks.length; i++) {
        data += '<tr>';
        data += '<td>' + (i + 1) + ". " + this.tasks[i] + '</td>';
        // Edit button
        data += '<td><button onclick="app.Edit(' + i + ')" class="btn btn-warning">DÃ¼zenle</button></td>';
        // Delete button
        data += '<td><button onclick="app.Delete(' + i + ')" class="btn btn-danger">Sil</button></td>';
        data += '</tr>';
      }
    }
    // Update the task count and display the task list
    this.Count(this.tasks.length);
    return this.el.innerHTML = data;
  };

  // Function to add a new task
  this.Add = function() {
    el = document.getElementById('add-todo');
    // Get the value
    var task = el.value;
    if (task) {
      // Add the new value
      this.tasks.push(task.trim());
      // Reset input value
      el.value = '';
      // Display the new list
      this.FetchAll();
    }
  };

  // Function to edit a task
  this.Edit = function(item) {
    var el = document.getElementById('edit-todo');
    // Display value in the field
    el.value = this.tasks[item];
    // Display fields
    document.getElementById('edit-box').style.display = 'block';
    var self = this;
    document.getElementById('save-edit').onsubmit = function() {
      // Get value
      var task = el.value;
      if (task) {
        // Edit value
        self.tasks.splice(item, 1, task.trim());
        // Display the new list
        self.FetchAll();
        // Hide fields
        CloseInput();
      }
    }
  };

  // Function to delete a task
  this.Delete = function(item) {
    // Delete the current task
    this.tasks.splice(item, 1);
    // Display the new list
    this.FetchAll();
  };

  // Function to update the task count
  this.Count = function(data) {
    var el = document.getElementById('counter');
    var name = 'Urun';
    if (data) {
      if (data == 1) {
        name = 'Urun';
      }
      el.innerHTML = data + ' ' + name;
    } else {
      el.innerHTML = name + ' yok ';
    }
  };
};

// Initial fetch and display of tasks
app.FetchAll();

// Function to close the task edit box
function CloseInput() {
  document.getElementById('edit-box').style.display = 'none';
}
